#include <bits/stdc++.h>
using namespace std;

class sudoku{
   public:
   int s[9][9];
   sudoku(){
      int a[9];
      for(int i=0;i<9;i++){
         cin>>a[i];
      }
      for(int i=0;i<9;i++){
         for(int j=8;j>=0;j--){
            s[i][j]=a[i]%10;
            a[i]/=10;
         }
      }
   }
   bool checker(int i,int j){
      for(int k=0;k<9;k++){
         if(k==j)continue;
         if(s[i][k]==s[i][j])return false;
      }
      for(int k=0;k<9;k++){
         if(k==i)continue;
         if(s[k][j]==s[i][j])return false;
      }
      for(int k=(i/3)*3;k<(i/3)*3+3;k++){
         for(int l=(j/3)*3;l<(j/3)*3+3;l++){
            if(k==i && l==j)continue;
            if(s[k][l]==s[i][j])return false;
         }
      }
      return true;
   }
   bool filler(int x=0,int y=0){
      for(int i=x;i<9;i++){
         for(int j=y;j<9;j++){
            if(s[i][j]==0){
               for(int k=1;k<=9;k++){
                  int c;
                  s[i][j]=k;
                  if(checker(i,j)){
                     for(int i1=i,j1=j;i1<9;i1++){
                        c=0;
                        if(i1>i)j1=0;
                        for(;j1<9;j1++){
                           if(s[i1][j1]==0){
                              if(!filler(i1,j1)){
                                 s[i][j]=0;
                                 c=1;
                              }
                              else return true;;
                              if(c==1 )break;
                           }
                        }
                        if(c==1)break;
                        if(i1+1>=9 && j1>=9)return true;
                     }
                  }
                  if(k==9){s[i][j]=0; return false;}
               }
            }
         }
      }
   }
};

int main() {
   sudoku s;
   s.filler();
   for(int i=0;i<9;i++){
      for(int j=0;j<9;j++){
         cout<<s.s[i][j];
      }
      cout<<endl;
   }
	return 0;
}
